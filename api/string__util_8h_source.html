142 template <typename T>
143 requires std::is_floating_point_v<T> && (!FromChars<T>)
144 static Result<T> ParseNumber(std::string_view str) {
145 T value{};
146 // strto* require null-terminated input; string_view does not guarantee it.
147 std::string owned(str);
148 const char* start = owned.c_str();
149 char* end = nullptr;
150 errno = 0;
151
152 if constexpr (std::same_as<T, float>) {
153 value = std::strtof(start, &end);
154 } else if constexpr (std::same_as<T, double>) {
155 value = std::strtod(start, &end);
156 } else {
157 value = std::strtold(start, &end);
158 }
159
160 if (end == start || end != start + static_cast<std::ptrdiff_t>(owned.size())) {
161 return InvalidArgument("Failed to parse {} from string '{}': invalid argument",
162 typeid(T).name(), str);
163 }
164 if (errno == ERANGE) {
165 return InvalidArgument("Failed to parse {} from string '{}': value out of range",
166 typeid(T).name(), str);
167 }
168 return value;
169 }

172 // ASCII-only case mappings. These avoid std::toupper/std::tolower, which are
173 // locale-dependent and have undefined behavior for negative char values.
174 static constexpr char ToUpperAscii(char c) noexcept {
175 return (c >= 'a' && c <= 'z') ? static_cast<char>(c - 'a' + 'A') : c;
176 }
177 static constexpr char ToLowerAscii(char c) noexcept {
178 return (c >= 'A' && c <= 'Z') ? static_cast<char>(c - 'A' + 'a') : c;
179 }
180
181 // True if c is a 7-bit ASCII byte (< 0x80). The cast is required because char may be
182 // signed, which would make bytes >= 0x80 compare as negative.
183 static constexpr bool IsAsciiByte(char c) noexcept {
184 return (static_cast<unsigned char>(c) & 0x80) == 0;
185 }
186
187 // Case-insensitive equality decided in a single byte-wise pass, without allocating.
188 // Returns nullopt once a byte of either operand is non-ASCII, because folding can then
189 // be non-ASCII and length-changing (e.g. "İ" (U+0130) -> "i"), which only ToLower
190 // knows.
191 static std::optional<bool> AsciiEqualsIgnoreCase(std::string_view a,
192 std::string_view b) {
193 const size_t n = std::min(a.size(), b.size());
194 for (size_t i = 0; i < n; ++i) {
195 if (!IsAsciiByte(a[i]) || !IsAsciiByte(b[i])) {
196 return std::nullopt;
197 }
198 if (ToLowerAscii(a[i]) != ToLowerAscii(b[i])) {
199 return false;
200 }
201 }
202 return a.size() == b.size();
203 }
