175 template <typename T>
176 using ByIdMap = std::unordered_map<int32_t, std::shared_ptr<T>>;
177 using SchemasMap = ByIdMap<Schema>;
178 using PartitionSpecsMap = ByIdMap<PartitionSpec>;
179 using SortOrdersMap = ByIdMap<SortOrder>;
180 using SnapshotsMap = std::unordered_map<int64_t, std::shared_ptr<Snapshot>>;
181 using SchemasMapRef = std::reference_wrapper<const SchemasMap>;
182 using PartitionSpecsMapRef = std::reference_wrapper<const PartitionSpecsMap>;
183 using SortOrdersMapRef = std::reference_wrapper<const SortOrdersMap>;
184 using SnapshotsMapRef = std::reference_wrapper<const SnapshotsMap>;
185
186 Result<SchemasMapRef> GetSchemasById() const;
187 Result<PartitionSpecsMapRef> GetPartitionSpecsById() const;
188 Result<SortOrdersMapRef> GetSortOrdersById() const;
189 Result<SnapshotsMapRef> GetSnapshotsById() const;
190
191 private:
192 static Result<SchemasMap> InitSchemasMap(const TableMetadata* metadata);
193 static Result<PartitionSpecsMap> InitPartitionSpecsMap(const TableMetadata* metadata);
194 static Result<SortOrdersMap> InitSortOrdersMap(const TableMetadata* metadata);
195 static Result<SnapshotsMap> InitSnapshotMap(const TableMetadata* metadata);
196
197 const TableMetadata* metadata_;
198 Lazy<InitSchemasMap> schemas_map_;
199 Lazy<InitPartitionSpecsMap> partition_specs_map_;
200 Lazy<InitSortOrdersMap> sort_orders_map_;
201 Lazy<InitSnapshotMap> snapshot_map_;
202};
