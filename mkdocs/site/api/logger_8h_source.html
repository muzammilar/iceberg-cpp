328template <typename... Args>
329struct FmtWithLoc {
330 std::format_string<Args...> fmt;
331 std::source_location loc;
332
333 template <typename T>
334 requires std::convertible_to<const T&, std::format_string<Args...>>
335 consteval FmtWithLoc( // NOLINT(google-explicit-constructor): mirrors
336 // std::format_string
337 const T& s, std::source_location loc = std::source_location::current())
338 : fmt(s), loc(loc) {}
339};

345template <typename... Args>
346void FormatAndEmit(Logger& logger, LogLevel level, const std::source_location& loc,
347 std::format_string<Args...> fmt, Args&&... args) noexcept {
348 if (!logger.ShouldLog(level)) return;
349 try {
350 Emit(logger, level, loc, std::format(fmt, std::forward<Args>(args)...));
351 } catch (...) {
352 // Catch-all upholds the noexcept "logging never throws" guarantee: a
353 // user-defined std::formatter may throw a non-std::exception type, and this
354 // function is noexcept, so anything escaping here would call std::terminate.
355 EmitFormatError(logger, level, loc);
356 }
357}
