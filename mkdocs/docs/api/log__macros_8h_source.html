61template <typename MakeMessage>
62void EmitIfEnabled(Logger& logger, LogLevel level, const std::source_location& location,
63 MakeMessage&& make_message) noexcept {
64 if (!logger.ShouldLog(level)) return;
65 try {
66 Emit(logger, level, location, std::forward<MakeMessage>(make_message)());

68 EmitFormatError(logger, level, location);

84template <typename MakeMessage>
85void LogToCurrentRuntime(LogLevel level, const std::source_location& location,
86 MakeMessage&& make_message) noexcept {
87 const std::shared_ptr<Logger>& logger = CurrentLogger();
88 if (logger) {
89 EmitIfEnabled(*logger, level, location, std::forward<MakeMessage>(make_message));
90 }
91 if (level == LogLevel::kFatal) {
92 if (logger) logger->Flush();
93 std::abort();

118template <typename MakeMessage>
119[[noreturn]] void LogFatal(const std::source_location& location,
120 MakeMessage&& make_message) noexcept {
121 std::string message;
122 try {
123 message = std::forward<MakeMessage>(make_message)();
124 } catch (...) {
125 message = "<fmt error>";
126 }
127 auto logger = GetCurrentLogger();
128 if (logger) {
129 if (logger->ShouldLog(LogLevel::kFatal)) {
130 Emit(*logger, LogLevel::kFatal, location, std::string(message));
131 }
132 logger->Flush();
133 }
134 if (auto handler = GetFatalHandler()) {
135 try {
136 handler(location, message);
137 } catch (...) { // a throwing handler must not prevent the abort
138 }
139 }
140 std::abort();
141}

145// ---------------------------------------------------------------------------
146// Logging macros.
147//
148// Every macro takes a std::format string followed by its arguments. The
149// rendered line depends on the active backend (see cerr_logger.h for the
150// std::cerr layout, or the spdlog pattern); the examples below show the call
151// site and, for the default CerrLogger, the line it produces.
152//
153// ICEBERG_LOG_TRACE("entering scan for {}", table);
154// 2026-06-16T10:59:41.186Z trace [12345] [table_scan.cc:88] entering scan for db.t
155// ICEBERG_LOG_DEBUG("cache miss key={}", key);
156// 2026-06-16T10:59:41.186Z debug [12345] [cache.cc:42] cache miss key=manifest-7
157// ICEBERG_LOG_INFO("loaded {} manifests in {} ms", n, ms);
158// 2026-06-16T10:59:41.186Z info [12345] [table_scan.cc:91] loaded 5 manifests in 12
159// ms
160// ICEBERG_LOG_WARN("retry {} after {}", attempt, err);
161// 2026-06-16T10:59:41.186Z warn [12345] [io.cc:51] retry 2 after timeout
162// ICEBERG_LOG_ERROR("commit failed: {}", status);
163// 2026-06-16T10:59:41.186Z error [12345] [txn.cc:77] commit failed: conflict
164// ICEBERG_LOG_CRITICAL("metadata unreadable at {}", path);
165// 2026-06-16T10:59:41.186Z critical [12345] [meta.cc:30] metadata unreadable at
166// s3://b/m.json
167// ICEBERG_LOG_FATAL("unrecoverable: {}", reason); // emits, flushes, then
168// std::abort()
169// 2026-06-16T10:59:41.186Z fatal [12345] [boot.cc:19] unrecoverable: bad config
170//
171// Less common forms:
172// ICEBERG_LOG(level, "level chosen at runtime: {}", x); // runtime severity
173// ICEBERG_LOG_TO(logger, level, "to an explicit logger {}", y);
174// ICEBERG_LOG_RUNTIME_FMT(level, fmt_string, args...); // non-literal format
175//
176// Include short_log_macros.h for bare aliases (LOG_INFO, ...). A format string is
177// mandatory; zero extra args is fine (ICEBERG_LOG_INFO("done")).
178// ---------------------------------------------------------------------------

191// A message-builder lambda that formats lazily (only invoked past ShouldLog by the
192// EmitIfEnabled helpers), so disabled logs never evaluate their arguments.
193#define ICEBERG_INTERNAL_LOG_MESSAGE(FMT_, ...) \
194 [&]() -> ::std::string { return ::std::format((FMT_)__VA_OPT__(, ) __VA_ARGS__); }
195
196// Fixed-severity emit with a compile-time floor (`if constexpr`) then the shared
197// current-logger path. Formatting happens only on the taken path and never throws.
198#define ICEBERG_INTERNAL_LOG(level_, FMT_, ...) \
199 do { \
200 if constexpr ((level_) >= ICEBERG_LOG_ACTIVE_LEVEL) { \
201 ::iceberg::internal::LogToCurrent( \
202 (level_), ::std::source_location::current(), \
203 ICEBERG_INTERNAL_LOG_MESSAGE(FMT_ __VA_OPT__(, ) __VA_ARGS__)); \
204 } \
205 } while (0)

220// FATAL: emit if enabled (never compile-stripped), then ALWAYS flush + abort.
221// Acquires the effective (scoped-or-default) logger ONCE so a concurrent
222// SetDefaultLogger cannot flush a different logger than it emitted to.
223#define ICEBERG_LOG_FATAL(FMT_, ...) \
224 ::iceberg::internal::LogFatal( \
225 ::std::source_location::current(), \
226 ICEBERG_INTERNAL_LOG_MESSAGE(FMT_ __VA_OPT__(, ) __VA_ARGS__))
227
228// Generic, runtime-level form against the default logger. No compile-time floor
229// (the level is not a constant). Aborts when level == kFatal.
230#define ICEBERG_LOG(level_, FMT_, ...) \
231 ::iceberg::internal::LogToCurrentRuntime( \
232 (level_), ::std::source_location::current(), \
233 ICEBERG_INTERNAL_LOG_MESSAGE(FMT_ __VA_OPT__(, ) __VA_ARGS__))
234
235// Generic form targeting an EXPLICIT logger (must be an lvalue Logger&). Honors
236// only that logger's ShouldLog. Aborts when level == kFatal.
237#define ICEBERG_LOG_TO(logger_, level_, FMT_, ...) \
238 ::iceberg::internal::LogToExplicitRuntime( \
239 (logger_), (level_), ::std::source_location::current(), \
240 ICEBERG_INTERNAL_LOG_MESSAGE(FMT_ __VA_OPT__(, ) __VA_ARGS__))

242// Runtime (non-literal) format string against the default logger. Aborts when

244#define ICEBERG_LOG_RUNTIME_FMT(level_, FMT_, ...) \

246 (level_), ::std::source_location::current(), [&]() -> ::std::string { \
247 return ::iceberg::internal::VFormat((FMT_)__VA_OPT__(, ) __VA_ARGS__); \
248 })
